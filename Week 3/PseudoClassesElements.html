  • e.g.
    • hover
    • active
    • focus

  • link (

    :link; visited

    )
  • userAction(

    :hover; :active; :focus

    )
  • forms (interfaces) (

    :enabled; :checked; :disabled

    )
  • structural/ positional pseudoClasses (

    :first-child; :last-child; :nth-child(); :only-child

    )
  • textual (

    :first-letter; :first-line

    )
  • positional/ generated (

    :before; :after

    )

    :selection

      "transition-property"; size/ color/ position/ ...

      )
    • "transition-duration" (time of change)
    • "transition-timing (smooth tr. (linear) or different)
